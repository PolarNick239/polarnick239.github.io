min_precision=0.9
max_recall=0.0
max_recall_info = 'no result'

def process(clf, X_data, name):
    global max_recall, max_recall_info
    results = clf.predict_proba(X_data)
    fpr, tpr, thresholds = roc_curve(Y[test], results[:, 1])
    auc_area = auc(fpr, tpr)
    ax1.plot(fpr, tpr, label=name+' (AUC-ROC={:.2f})'.format(auc_area))
    
    precisions = np.zeros_like(thresholds)
    recalls = np.zeros_like(thresholds)
    for i, threshold in enumerate(thresholds):
        y_pred = (results[:, 1] >= threshold)
        TP = ((Y[test] == 1) & (y_pred == 1)).sum()
        FP = ((Y[test] == 0) & (y_pred == 1)).sum()
        FN = ((Y[test] == 1) & (y_pred == 0)).sum()
        # Precision = TP / (TP + FP + 1)  ,    ,   
        # Recall = TP / (TP + FN + 1)  ,  ,   
        precisions[i] = TP / (TP + FP + 1)
        recalls[i] = TP / (TP + FN + 1)
    precisions[0] = 1
    recalls[0] = 0
    
    candidate = recalls[precisions >= min_precision].max()
    if candidate >= max_recall:
        max_recall = candidate
        threshold = thresholds[recalls[precisions >= min_precision].argmax()]
        max_recall_info = name + ' with threshold={:.2f}'.format(threshold)
    ax2.plot(recalls, precisions, label=name)

def calculate_auc_roc(clf, X, Y):
    results = clf.predict_proba(X)
    fpr, tpr, thresholds = roc_curve(Y, results[:, 1])
    auc_area = auc(fpr, tpr)
    return auc_area

    return DecisionTreeClassifier(criterion=criterion, max_depth=max_depth, min_samples_split=min_samples_split,
                                  min_samples_leaf=min_samples_leaf).fit(X, Y)

def process(clf, X_data, name):
    global max_recall, max_recall_info
    results = clf.predict_proba(X_data)
    fpr, tpr, thresholds = roc_curve(Y[test], results[:, 1])
    auc_area = auc(fpr, tpr)
    ax1.plot(fpr, tpr, label=name+' (AUC-ROC={:.2f})'.format(auc_area))
    
    precisions = np.zeros_like(thresholds)
    recalls = np.zeros_like(thresholds)
    for i, threshold in enumerate(thresholds):
        y_pred = (results[:, 1] >= threshold)
        TP = ((Y[test] == 1) & (y_pred == 1)).sum()
        FP = ((Y[test] == 0) & (y_pred == 1)).sum()
        FN = ((Y[test] == 1) & (y_pred == 0)).sum()
        # Precision = TP / (TP + FP + 1)  ,    ,   
        # Recall = TP / (TP + FN + 1)  ,  ,   
        precisions[i] = TP / (TP + FP + 1)
        recalls[i] = TP / (TP + FN + 1)
    precisions[0] = 1
    recalls[0] = 0
    
    candidate = recalls[precisions >= min_precision].max()
    if candidate >= max_recall:
        max_recall = candidate
        threshold = thresholds[recalls[precisions >= min_precision].argmax()]
        max_recall_info = name + ' with threshold={:.2f}'.format(threshold)
    ax2.plot(recalls, precisions, label=name)
