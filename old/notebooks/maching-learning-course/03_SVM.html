class LinearSVM(BaseEstimator):
    def __init__(self, C, transform=None):
        self.C = C
        if transform is None:
            transform = lambda x: x
        self.transform = transform
        
    def fit(self, X, y):
        X = self.transform(X)
        n, dim = X.shape
        
        self.bias_ = 0 # x0
        self.weights_ = np.zeros(dim) # x1, x2, x3, ... x(dim-1), xdim
        self.slack_variables = np.zeros(n) # x(dim+1), x(dim+2), ..., x(dim+n-1), x(dim+n)
        
        P = np.diag([0] + [1.0] * dim + [0] * n)
        q = np.array([0] * (1 + dim) + [self.C] * n)
        
        # slack_variables >= 0
        G1 = np.hstack((np.zeros((n, (1 + dim))), np.diag([-1.0]*n)))
        h1 = np.array([0.0]*n)
        
        # yn(wT*xn+b)1n
        # -yn(wT*xn+b) <= n-1
        # -yn(wT*xn+b)-n <= -1
        # -(yn(wT*xn+b)+n) <= -1
        G2list = []
        for i in range(0, n):
            # yn(wT*xn+b)+n
            yn_b = np.array([y[i]])
            yn_w_xn = y[i] * X[i]
            slack_vars = np.zeros(n)
            slack_vars[i] = 1.0
            G2list.append(np.hstack((yn_b, yn_w_xn, slack_vars)))
        G2 = -np.array(G2list)
        h2 = np.array([-1.0] * n)
        
        G = np.vstack((G1, G2))
        h = np.hstack((h1, h2))
        
        #           (66 .  720 .)
        solution = cvx.solvers.qp(cvx.sparse(cvx.matrix(P)), cvx.matrix(q), cvx.sparse(cvx.matrix(G)), cvx.matrix(h))
        xs = np.array(solution['x']).T[0]
        
        self.bias_ = xs[0]
        self.weights_ = xs[1:(dim+1)]
        self.slack_variables = xs[(dim+1):(dim+n+1)]

        #      
        y_predict = self.predict_proba(X)
        tmp = np.sum(self.weights_*X, axis=1)
        vals = y_predict * (tmp + self.bias_) + self.slack_variables - 1.0
        EPSILON = 0.000001
        self.support_mask = vals <= EPSILON
        
    def predict_proba(self, X):
        X = self.transform(X)
        y_predict = (np.dot(X, self.weights_) + self.bias_).flatten()
        return y_predict
            
    def predict(self, X):
        result = np.sign(self.predict_proba(X))
        result[result == 0] = 1.0
        return result

def visualize(clf, X, y, quality=400):
    X0 = X[:, 0]
    X1 = X[:, 1]
    
    border = .2
    X0range = X0.max() - X0.min()
    X1range = X1.max() - X1.min()
    X0from, X0to = X0.min() - X0range * border, X0.max() + X0range * border
    X1from, X1to = X1.min() - X1range * border, X1.max() + X1range * border
    x0vals = np.linspace(X0from, X0to, quality)
    x1vals = np.linspace(X1from, X1to, quality)

    xs0, xs1 = np.meshgrid(x0vals, x1vals)
    ys = np.reshape(clf.predict_proba(np.vstack((xs0.ravel(), xs1.ravel())).T), xs0.shape)

    #  ,    
    plt.pcolormesh(xs0, xs1, ys, cmap='autumn', zorder=0)

    #  
    matplotlib.rcParams['contour.negative_linestyle'] = 'solid'
    plt.contour(xs0, xs1, ys, levels=(0.0,), colors='k', linewidths=1.5, zorder=1)
    plt.contour(xs0, xs1, ys, levels=(-1.0,), colors='k', linewidths=0.5, zorder=1)
    plt.contour(xs0, xs1, ys, levels=(1.0,), colors='k', linewidths=0.5, zorder=1)

    # yn(wTxn+b)=1n    
    y_predict = clf.predict(X)
    support_mask = clf.support_mask.copy()
    wrong_mask = y_predict != y
    support_mask[wrong_mask] = False
    others_mask = np.ones_like(support_mask, np.bool)
    others_mask[support_mask] = False
    others_mask[wrong_mask] = False
    
    def draw_points(mask, marker, size):
        pos_mask = y > 0
        mask1 = np.logical_and(mask, pos_mask)
        plt.scatter(X0[mask1], X1[mask1], c='#ffff00', marker=marker, s=size)
        mask2 = np.logical_and(mask, ~pos_mask)
        plt.scatter(X0[mask2], X1[mask2], c='r', marker=marker, s=size)
    
    draw_points(others_mask, 's', 10)
    draw_points(support_mask, 'o', 20)
    draw_points(wrong_mask, 'x', 20)
    
    plt.xlim([X0from, X0to])
    plt.ylim([X1from, X1to])

#   SVM

def kernel_linear():
    def foo(X1, X2):
        return np.dot(X1, X2)
    return foo

def kernel_polynomial(d=2):
    def foo(X1, X2):
        return (np.dot(X1, X2) + 1) ** d
    return foo

def kernel_gauss_RBF(sigma=1.0):
    def foo(X1, X2):
        norm2 = np.linalg.norm(X1-X2)**2
        return np.exp(-norm2/(2.*sigma))
    return foo

class KernelSVM(BaseEstimator):
    def __init__(self, C, kernel=kernel_linear()):
        self.C = C
        self.kernel = kernel
        
    def fit(self, X, y):
        n, dim = X.shape
        
        self._n, self._dim = n, dim
        self._y = y
        self._X = X
        
        self.bias_ = 0 # x0
        self.weights_ = np.zeros(dim) # x1, x2, x3, ... x(dim-1), xdim
        self.slack_variables = np.zeros(n) # x(dim+1), x(dim+2), ..., x(dim+n-1), x(dim+n)
        
        K = np.zeros((n, n), np.float)
        P = np.zeros((n, n), np.float)
        for i in range(0, n):
            for j in range(0, n):
                K[i, j] = self.kernel(X[i, :], X[j, :])
                P[i, j] = y[i] * y[j] * K[i, j]
        q = np.array([-1.] * n)
        
        G = np.vstack((np.diag([1.]*n), np.diag([-1.]*n)))
        h = np.array([self.C]*n + [0.]*n)
        
        A = np.array([y.T], np.float)
        b = np.array([0.])
        
        #           (66 .  720 .)
        solution = cvx.solvers.qp(cvx.sparse(cvx.matrix(P)), cvx.matrix(q), cvx.sparse(cvx.matrix(G)), cvx.matrix(h),
                                  cvx.matrix(A), cvx.matrix(b))
        self._xs = np.array(solution['x']).T[0]
        
        EPSILON = 0.000001
        self.support_mask = self._xs > EPSILON
        
        self._biases = []
        for i in range(0, n):
            bias = y[i] - self._xs[i]*y[i]*np.sum(K[:, i])
            self._biases.append(bias)
        self.bias = np.average(self._biases)
        
    def predict_proba(self, X):
        m = X.shape[0]
        proba = np.zeros(m)
        for i in range(0, m):
            for j in range(0, self._n):
                proba[i] += self._xs[j] * self._y[j] * self.kernel(X[i, :], self._X[j, :])
            proba[i] += self.bias
        return proba
            
    def predict(self, X):
        result = np.sign(self.predict_proba(X))
        result[result == 0] = 1.0
        return result
